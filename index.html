<head>
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>新春贪吃蛇</title>
    <style>
        body { margin: 0; touch-action: none; background: #ff0000; }
        #gameCanvas { background: #cc0000; }
        .controls {
            position: fixed;
            bottom: 20px;
            width: 100%;
            text-align: center;
        }
        .control-btn {
            width: 60px;
            height: 60px;
            margin: 5px;
            background: rgba(255, 215, 0, 0.3);
            border: 2px solid #FFD700;
            border-radius: 50%;
        }
    </style>
</head>
<body>
<canvas id="gameCanvas"></canvas>
<div class="controls">
    <button class="control-btn" onclick="changeDirection('UP')">↑</button><br>
    <button class="control-btn" onclick="changeDirection('LEFT')">←</button>
    <button class="control-btn" onclick="changeDirection('RIGHT')">→</button><br>
    <button class="control-btn" onclick="changeDirection('DOWN')">↓</button>
</div>

<script>
const canvas = document.getElementById('gameCanvas');
const ctx = canvas.getContext('2d');

// 设置画布尺寸
function resizeCanvas() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
}
window.addEventListener('resize', resizeCanvas);
resizeCanvas();

// 新春祝福像素文字
function drawFestivalText() {
    ctx.fillStyle = 'rgba(255, 215, 0, 0.2)';
    ctx.font = '24px "Press Start 2P"';
    ctx.textAlign = 'center';
    ctx.fillText('新年快乐', canvas.width/2, canvas.height/3);
    ctx.fillText('万事如意', canvas.width/2, canvas.height/3 + 40);
}

// 游戏参数
let snake = [{x: 10, y: 10}];
let food = {x: 15, y: 15};
let direction = 'RIGHT';
let gridSize = 20;
let gameLoop;

// 生成食物
function generateFood() {
    food.x = Math.floor(Math.random() * (canvas.width/gridSize));
    food.y = Math.floor(Math.random() * (canvas.height/gridSize));
}

// 绘制渐变金色
function drawGold(x, y) {
    const gradient = ctx.createRadialGradient(
        x*gridSize + gridSize/2,
        y*gridSize + gridSize/2,
        0,
        x*gridSize + gridSize/2,
        y*gridSize + gridSize/2,
        gridSize/2
    );
    gradient.addColorStop(0, '#FFD700');
    gradient.addColorStop(1, '#D4AF37');
    return gradient;
}

// 游戏主循环
function gameStep() {
    // 移动蛇头
    let head = {...snake[0]};
    switch(direction) {
        case 'UP': head.y--; break;
        case 'DOWN': head.y++; break;
        case 'LEFT': head.x--; break;
        case 'RIGHT': head.x++; break;
    }

    // 碰撞检测
    if (head.x < 0 || head.x >= canvas.width/gridSize ||
        head.y < 0 || head.y >= canvas.height/gridSize ||
        snake.some(segment => segment.x === head.x && segment.y === head.y)) {
        clearInterval(gameLoop);
        alert('游戏结束！新春大吉！');
        return;
    }

    snake.unshift(head);

    // 吃食物检测
    if (head.x === food.x && head.y === food.y) {
        generateFood();
    } else {
        snake.pop();
    }

    // 绘制
    ctx.fillStyle = '#cc0000';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    drawFestivalText();

    // 绘制食物
    ctx.fillStyle = drawGold(food.x, food.y);
    ctx.beginPath();
    ctx.arc(
        food.x*gridSize + gridSize/2,
        food.y*gridSize + gridSize/2,
        gridSize/2, 0, Math.PI*2
    );
    ctx.fill();

    // 绘制蛇
    snake.forEach((segment, index) => {
        ctx.fillStyle = drawGold(segment.x, segment.y);
        ctx.fillRect(
            segment.x*gridSize,
            segment.y*gridSize,
            gridSize - 1,
            gridSize - 1
        );
    });
}

// 方向控制
function changeDirection(newDirection) {
    const opposite = {
        UP: 'DOWN', DOWN: 'UP',
        LEFT: 'RIGHT', RIGHT: 'LEFT'
    };
    if (newDirection !== opposite[direction]) {
        direction = newDirection;
    }
}

// 触控和键盘事件
document.addEventListener('keydown', (e) => {
    const directions = {
        ArrowUp: 'UP',
        ArrowDown: 'DOWN',
        ArrowLeft: 'LEFT',
        ArrowRight: 'RIGHT'
    };
    if (directions[e.key]) changeDirection(directions[e.key]);
});

// 开始游戏
generateFood();
gameLoop = setInterval(gameStep, 150);

// 禁止页面滚动
document.addEventListener('touchmove', (e) => e.preventDefault(), { passive: false });
</script>
</body>
</html>
